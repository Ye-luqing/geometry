\begin{exercise}
  Let $R$ be the region between the (polar) graphs of $r=f_1(\theta)$
  and $r=f_2(\theta)$,where $a\leq\theta\leq b$.Find a
  parameterization for $R$.
\end{exercise}
\begin{proof}[Solve]
$([tf_1(\theta)+(1-t)f_2(\theta)]\cos\theta,[tf_1(\theta)+(1-t)f_2(\theta)]\sin\theta)$,where $\theta\in [a,b],t\in [0,1]$.  
\end{proof}
