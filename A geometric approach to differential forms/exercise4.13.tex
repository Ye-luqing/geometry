\begin{exercise}
  Let $C$ be the curve in $\mathbf{R}^2$ given by $\psi(t)=(t\cos
  t,t\sin t)$,where $0\leq t\leq 2$.Let $\nu =-ydx+xdy$.Compute
  $\int_C\nu$(with the induced orientation).
\end{exercise}
\begin{proof}[Solve]
  \begin{align*}
    \int_C\nu &=\int_{[1,2]}\nu(\frac{\pa \psi}{\pa t})dt
\\&=\int_{[1,2]}\nu(\cos t-t\sin t,\sin t+t\cos t)dt
\\&=\int_{[1,2]}-(\sin t+t\cos t)(\cos t-t\sin t)+(\cos t-t\sin
t)(\sin t+t\cos t)dt
\\&=0.
  \end{align*}
\end{proof}
