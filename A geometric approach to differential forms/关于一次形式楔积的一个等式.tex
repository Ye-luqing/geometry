\title{\huge{\bf{关于一次形式楔积的一个等式}}} \author{\small{叶卢庆\footnote{叶卢庆(1992---),男,杭州师范大学理学院数学与应用数学专业本科在读,E-mail:yeluqingmathematics@gmail.com}}}
\maketitle
设
$w=a_1dx_1+a_2dx_2+\cdots+a_ndx_n$,$v=b_1dx_1+b_2dx_2+\cdots+b_ndx_n$.且
向量 $V_1=(c_1,c_2,\cdots,c_{n}),V_2=(d_1,d_{2},\cdots,d_{n})\in T_p\mathbf{R}^n$.且$\langle w\rangle
=(a_1,a_2,\cdots,a_n)$,$\langle
v\rangle=(b_1,b_2,\cdots,b_n)$,$V_1^{-1}=c_1dx_1+c_2dx_2+\cdots+c_ndx_n$,$V_2^{-1}=d_1dx_1+d_2dx_2+\cdots+d_ndx_n$.则
我们有
$$
w\wedge v(V_1,V_2)=V_1^{-1}\wedge V_2^{-1}(\langle w\rangle,\langle v\rangle).
$$
