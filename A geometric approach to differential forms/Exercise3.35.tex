\begin{exercise}
  Let $\omega$ be an $n$-form and $\nu$ an $m$-form.Show that 
$$
\omega\wedge \nu=(-1)^{nm}\nu\wedge \omega.
$$
\end{exercise}
\begin{proof}
  Without loss of generality we assume that $\omega=dx_1\wedge
  dx_2\cdots\wedge dx_n$,and $\nu=dy_1\wedge dy_2\wedge\cdots\wedge
  dy_m$.Then 
$$
\omega\wedge\nu=dx_1\wedge
  dx_2\cdots\wedge dx_n\wedge dy_1\wedge dy_2\wedge\cdots\wedge
  dy_m.
$$

\end{proof}
