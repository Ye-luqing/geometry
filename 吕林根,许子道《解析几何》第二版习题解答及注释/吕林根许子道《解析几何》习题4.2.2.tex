\begin{exercise}
  已知锥面的顶点为$(3,-1,-2)$,准线为$x^2+y^2-z^2=1$,$x-y+z=0$,试求其方
  程.
\end{exercise}
\begin{proof}[\textbf{解}]
设锥面上任意一点$(x_{0},y_{0},z_{0})$,则存在实数$t$,使得
$$
(x_0,y_0,z_0)+t(x_0-3,y_0+1,z_0+2)
$$
在锥面上.于是,
$$
\begin{cases}
  (x_0+t(x_0-3))^2+(y_0+t(y_0+1))^2-(z_0+t(z_0+2))^2=1,\\
x_0+t(x_0-3)-y_0-t(y_0+1)+z_0+t(z_0+2)=0.
\end{cases}
$$
消去$t$,可得
$$
7z_{0}^2-5y_0^2+3x_0^2+10x_{0}z_{0}-6x_{0}y_{0}-2y_{0}z_{0}-4x_{0}+4y_{0}-4z_{0}+4=0.
$$
于是锥面方程为
$$
7z^2-5y^2+3x^2+10xz-6xy-2yz-4x+4y-4z+4=0.
$$
\end{proof}
