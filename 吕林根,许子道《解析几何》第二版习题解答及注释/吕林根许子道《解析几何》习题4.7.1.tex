\begin{exercise}
  求下列直纹曲面的直母线族方程:
  \begin{itemize}
  \item $x^2-y^2-z^2=0$.
\item $z=axy$.
  \end{itemize}
\end{exercise}
\begin{itemize}
\item 
$$
x^2-y^2-z^2=0 \iff
\begin{cases}
  v(x+z)=uy\\
u(x-z)=vy
\end{cases}
$$
其中$v,u$是参数且不同时为$0$.
\item 
$$
z=axy \iff
\begin{cases}
  z=aux,\\
u=y.
\end{cases}\mbox{或}
\begin{cases}
  z=avy,\\
v=x
\end{cases}.
$$
其中$u,v$为参数.
\end{itemize}
