\begin{exercise}
  将下列平面的一般方程化为法式方程:
  \begin{itemize}
  \item $x-2y+5z-3=0$.
\item $x-y+1=0$.
\item $x+2=0$.
\item $4x-4y+7z=0$.
  \end{itemize}
\end{exercise}
\begin{itemize}
\item 
$$
x-2y+5z-3=0\ri \frac{1}{\sqrt{30}}x-\frac{2}{\sqrt{30}}y+\frac{5}{\sqrt{30}}z-\frac{3}{\sqrt{30}}=0.
$$
\item $$x-y+1=0\ri
  -\frac{1}{\sqrt{2}}x+\frac{1}{\sqrt{2}}y-\frac{1}{\sqrt{2}}=0.$$
\item $$ -x-2=0 .$$
\item $$ \frac{4}{9}x-\frac{4}{9}y+\frac{7}{9}z=0. $$
\end{itemize}
